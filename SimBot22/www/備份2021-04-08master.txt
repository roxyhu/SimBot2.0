    let change_mark_txt = ''
    let change_period = /[.]/g;
    change_mark_txt = take_word_dialog_talkroom.replace(change_period, ".<br>")
    let change_question_mark = /[?]/g;
    change_mark_txt = change_mark_txt.replace(change_question_mark, "?<br>")
    let change_Exclamation_mark = /[!]/g;
    change_mark_txt = change_mark_txt.replace(change_Exclamation_mark, "!<br>")
    if (change_mark_txt.indexOf("Student:") >= 0) {
      // let change_colon_mark = /[:]/g;
      // change_mark_txt = change_mark_txt.replace(change_colon_mark, ":")
      change_mark_txt = change_mark_txt.split("Student:")

      $('#p_val').prepend(`<div class="bot"align="left"><span>${user}:<br>${change_mark_txt[0]}</br></span></div>`)
    }
    else {
      $('#p_val').prepend(`<div class="bot"align="left"><span>${user}:<br>${change_mark_txt}</br></span></div>`)
    }

  }

  //(文字比對)如果跟使用者與機器辨識的字去比對有相同的用紅字顯示
  function word_compare(similar_words, inputdom) {
    //let test1='this is so green black yellow'; //original word
    //let test2='green is so light blue'; //user speak word
    let similar_words_temp1 = similar_words.split(" ")
    let inputdom_temp2 = inputdom.split(" ")
    let change_word_color = []
    let word_total = ""
    let change_count = 0
    let no_change_count = 0


    word_compare_length(similar_words_temp1, inputdom_temp2)
    for (i = 0; i < similar_words_temp1.length; i++) {
      for (j = 0; j < inputdom_temp2.length; j++) {
        // console.log(temp2[i].indexOf(temp1[j]));
        if (inputdom_temp2[i].indexOf(similar_words_temp1[j]) == 0) {
          change_word_color[change_count] = `<font color="#000000">${inputdom_temp2[i]}</font>`;
          change_count += 1
        }
        else {
          no_change_count += 1
          if (no_change_count == similar_words_temp1.length) {
            change_word_color[change_count] = `<font color="#FF0000">${inputdom_temp2[i]}</font>`;
            no_change_count = 0
            change_count += 1
          }
        }
      }
      no_change_count = 0
      word_total += change_word_color[i] + " "
    }
    //  console.log(change_word_color)
    //  console.log(word_total)
    return word_total;
  }
  //(文字比對)長度不同會無法比較(前處理)
  function word_compare_length(similar_words_temp1, inputdom_temp2) {
    let count_length_1 = similar_words_temp1.length
    let count_length_2 = inputdom_temp2.length
    if (similar_words_temp1.length > inputdom_temp2.length) {
      for (i = 0; i < (similar_words_temp1.length - inputdom_temp2.length); i++) {
        inputdom_temp2[count_length_2] = ""
        count_length_2 += 1
      }
    }
    else if (similar_words_temp1.length < inputdom_temp2.length) {
      for (i = 0; i < (inputdom_temp2.length - similar_words_temp1.length); i++) {
        similar_words_temp1[count_length_1] = ""
        count_length_1 += 1
      }
    }
    //console.log(temp1,temp2)
  }
